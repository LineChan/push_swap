31 static void ft_printf_find_flags(char c, t_struct *arg)
32 {
33  if (c == '-')
34  arg->flag |= FLAG_MINUS;
35  else if (c == '+')
36  arg->flag |= FLAG_PLUS;
37  else if (c == '0' && !DOT && !arg->padd)
38  arg->flag |= FLAG_ZERO;

40  arg->flag |= FLAG_SPACE;
41  else if (c == '.')
42  arg->flag |= FLAG_DOT;
43  else if (!DOT && ft_isdigit(c))
44  arg->padd = arg->padd * 10 + c - 48;

47  else if (c == '#')
